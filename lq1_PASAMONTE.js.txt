let persona1Name = "Maria Delos Santos Las Marias";
let persona1Age = 25;
let persona1Address = "Upper Session Road, Baguio City, Benguet";

let persona2Name = "Juan Gamoso Dela Cruz";
let persona2Age = 28;
let persona2Address = "San Nicolas, Candon City, Ilocos Sur";

// Using split to rearrange names into the format LASTNAME (all caps), First Name Middle Name
function rearrangeName(fullName) {
    let nameParts = fullName.split(" ");
    let lastName = nameParts.pop().toUpperCase(); //last name, converted to uppercase
    let firstName = nameParts.shift(); // first name
    let middleName = nameParts.join(" "); // Join remaining parts as the middle name
    return `${lastName}, ${firstName} ${middleName}`;
}

let rearrangedPersona1Name = rearrangeName(persona1Name);
let rearrangedPersona2Name = rearrangeName(persona2Name);

console.log("Name:"+`${rearrangedPersona1Name}, ${persona1Age}, ${persona1Address}`);
console.log("Name:"+`${rearrangedPersona2Name}, ${persona2Age}, ${persona2Address}`);

let persona1NameLength = persona1Name.length;
let persona2NameLength = persona2Name.length;
let persona1AddressLength = persona1Address.length;
let persona2AddressLength = persona2Address.length;

console.log(persona1NameLength);
console.log(persona2NameLength);
console.log(persona1AddressLength);
console.log(persona2AddressLength);

// 1. Sum of all numeric values from var declarations (ages and lengths of names and addresses)
let totalSum = persona1Age + persona2Age + persona1NameLength + persona2NameLength + persona1AddressLength + persona2AddressLength;
console.log(totalSum);

// 2. Add the ages of both personas, subtract the length of Persona 1’s name from the result, then multiply by the length of Persona 2’s name.
let result = (persona1Age + persona2Age - persona1NameLength) * persona2NameLength;
console.log(result);

// 3. Raise the length of Persona 1’s address to the power of Persona 2’s address length.
